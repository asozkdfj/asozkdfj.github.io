#include <iostream>
#include <vector>
using namespace std;
int main(void) {
    int loop_num;
    cin >> loop_num;

    for (int i = 0; i < loop_num; i++) {
        int ppl_num;
        cin >> ppl_num;
        double avg = 0;
        double sc;
        vector<int> score;

        for (int j = 0; j < ppl_num; j++) {
            cin >> sc;
            score.push_back(sc);
            avg += sc;
        }
        avg /= ppl_num;
        double cnt = 0;
        for (int j = 0; j < ppl_num; j++) {
            if (avg < score[j]) {
                cnt++;
            }
        }

        cout << fixed;          //소숫점을 고정해서 표현하겠다.
        cout.precision(3);      //소숫점 자리 표시
        cout << cnt/ppl_num*100 << "%" << endl;
    }
    return 0;
}
